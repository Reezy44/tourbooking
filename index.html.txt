        function moveSlider(direction) {
            const itemsPerView = window.innerWidth >= 1024 ? 3 : window.innerWidth >= 768 ? 2 : 1;
            const maxSlide = Math.ceil(destinations.length / itemsPerView) - 1;

            if (currentSlide > maxSlide) currentSlide = 0;
            
            updateSlider();
        }

            currentSlide = index;

        function updateSlider() {
            const slider = document.getElementById('destination-slider');
            const itemsPerView = window.innerWidth >= 1024 ? 3 : window.innerWidth >= 768 ? 2 : 1;
            const translate = currentSlide * (100 / itemsPerView);
            slider.style.transform = `translateX(-${translate}%)`;
            
            // Update dots
            const dots = document.querySelectorAll('#slider-dots button');
            dots.forEach((dot, i) => {
                if (i === currentSlide) {
                    dot.classList.remove('bg-gray-300', 'w-3');
                    dot.classList.add('bg-lk-primary', 'w-8');
                } else {
                    dot.classList.remove('bg-lk-primary', 'w-8');
                    dot.classList.add('bg-gray-300', 'w-3');
                }
            });
        }
